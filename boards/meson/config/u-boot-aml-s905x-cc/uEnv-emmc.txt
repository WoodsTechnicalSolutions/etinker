# 32 GiB eMMC on MMC 1
uenvcmd=mmc rescan; mmc dev 0 && \
echo && echo Booting 'etinker' environment ... && echo && \
setenv bootargs console=ttyAML0,115200 root=/dev/mmcblk1p1 rw rootfstype=ext4 rootwait && \
ext2load mmc 1:2 $kernel_addr_r boot/Image && \
ext2load mmc 1:2 $fdt_addr_r boot/meson-gxl-s905x-libretech-cc.dtb && \
booti $kernel_addr_r - $fdt_addr_r
